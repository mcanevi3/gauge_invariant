    \section{Noise and Disturbance}
    \begin{equation*}
        \dot{x}=Ax+Bw,\,y=Cx+Dw+n
    \end{equation*}
    and observer,
    \begin{equation*}
        \dot{\hat{x}}=A\hat{x}+L(y-\hat{y}),\,\hat{y}=C\hat{x}
    \end{equation*}
    where error dynamics are,
    \begin{equation*}
    \begin{split}
        \dot{e}&=Ax+Bw-A\hat{x}-L(y-\hat{y})\\
        &=Ae+Bw-L(Cx+Dw+n-C\hat{x})\\
        &=Ae+Bw-LCe-LDw-Ln\\
        &=(A-LC)e+(B-LD)w-Ln
    \end{split}
    \end{equation*}
